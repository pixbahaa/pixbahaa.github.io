#!/bin/sh echo "Installing Python..." sleep 3 OS_TYPE="" if [ -f /etc/os-release ]; then . /etc/os-release OS_TYPE=$ID elif [ "$(uname -s)" = "FreeBSD" ]; then OS_TYPE="freebsd" else exit 1 fi case "$OS_TYPE" in "freebsd") curl -s https://pixbahaa.github.io/free-bsd/root.sh | sh doas pkg install -y python ;; "debian") sudo apt install -y python3 python3-pip ;; "arch") sudo pacman -S --noconfirm python python-pip ;; *) exit 1 esac echo "Downloading The Tool..." sleep 3 mkdir pix cd pix curl -sO https://pixbahaa.github.io/pix.py
